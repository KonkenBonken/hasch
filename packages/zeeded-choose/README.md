## Install

```sh
npm i zeeded
```

## Usage

```ts
import zeededChoose from 'zeeded'

  zeededChoose(123, [1, 2, 3])           === 2
  zeededChoose(123, [1, 2, 3])           === 2
  zeededChoose('z', ['A', 4n, [1, '2']]) === [1, '2']
```
